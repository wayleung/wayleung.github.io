【其实还有最简单的方法不需要反射，****枚举类中有一个静态方法values()，value()方法可以将枚举类转变为一个枚举类型的数组】

/**
     *  判断枚举值是否存在
     */
    public static boolean isExist(Integer value) {
        if (value == null) {
            return false;
        }
        for (AppIdEnum e : values()) {
            if (value.equals(e.getAppId())) {
                return true;
            }
        }
        return false;
    }
